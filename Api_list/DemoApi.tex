2. Retrieve the list of users who have made more than 3 orders.

[
  {
    $unwind: {
      path: "$products",
    }
  },
  {
    $group: {
      _id: "$_id",
      "CountOrder": {
        $sum: "$products.quantity"
      }
    }
  },
  {
    $match: {
      "CountOrder":{$gt : 3}
    }
  }
]
